        \phantomsection
        \includegraphics[width=#1]{#2}

        \captionof{figure}{#3}
        \label{#4}
    \end{center}
}



\newcommand{\newcode}[5][1.0]{
    \vspace{0.1cm}
    \begin{center}
        \captionsetup{hypcap=false} % <-- این خط هشدار را فقط برای این کد غیرفعال می‌کند
        \resizebox{#1\columnwidth}{!}{%
            \parbox{\linewidth}{%
                \ifpersiandoc
                    \begin{latin}
                        \lstinputlisting[language=#2]{#3}
                    \end{latin}
                \else
                    \lstinputlisting[language=#2]{#3}
                \fi
            }%
        }
        \vspace{-0.4cm}
        \captionof{lstlisting}{#4}
        \label{#5}
    \end{center}
    \vspace{0.1cm}
}
